      # Quick edits
      conf = "cd ~/.config";
      nixconf = "cd /etc/nixos";
      
      # SSH with proper TERM
      ssh = "env TERM=xterm-256color ssh";
    };
    
    # Interactive shell init
    interactiveShellInit = ''
      # Disable greeting
      set -g fish_greeting
      
      # Initialize starship
      ${pkgs.starship}/bin/starship init fish | source
      
      # Initialize zoxide
      ${pkgs.zoxide}/bin/zoxide init fish | source
      
      # Set color variables (Sonokai theme)
      set -gx BLACK 0xff181819
      set -gx WHITE 0xffe2e2e3
      set -gx RED 0xfffc5d7c
      set -gx GREEN 0xff9ed072
      set -gx BLUE 0xff76cce0
      set -gx YELLOW 0xffe7c664
      set -gx ORANGE 0xfff39660

          builtin cd $argv
          and eza --icons
      end
      
      # Quick navigation with zoxide
      function z
          __zoxide_z $argv
          and eza --icons
      end

      function extract
          switch $argv[1]
              case '*.tar.bz2'
                  tar xjf $argv[1]
              case '*.tar.gz'
                  tar xzf $argv[1]
              case '*.bz2'
                  bunzip2 $argv[1]
              case '*.rar'
                  unrar x $argv[1]
              case '*.gz'
                  gunzip $argv[1]
              case '*.tar'
                  tar xf $argv[1]
              case '*.tbz2'
                  tar xjf $argv[1]
              case '*.tgz'
                  tar xzf $argv[1]
              case '*.zip'
                  unzip $argv[1]
              case '*.Z'
                  uncompress $argv[1]
              case '*.7z'
                  7z x $argv[1]
              case '*'
                  echo "'$argv[1]' cannot be extracted"
          end
      end
      
      # Keybindings
      bind \cf 'zi'  # Ctrl+F for interactive zoxide
    '';
  };

    enableCompletion = true;

Script to toggle between workspace and menu view
#!/usr/bin/env bash
# Toggle between workspace view and app menu view in waybar

STATE_FILE="/tmp/waybar-menu-state"
CONFIG_DIR="$HOME/.config/waybar"

# Check current state
if [ -f "$STATE_FILE" ]; then
    STATE=$(cat "$STATE_FILE")
else
    STATE="workspaces"
fi

# Toggle state
if [ "$STATE" = "workspaces" ]; then
    # Switch to menu view
    echo "menu" > "$STATE_FILE"
    
    # Get current app
    APP=$(hyprctl activewindow -j | jq -r '.class')
    
    # Update waybar config to show menus
    cat > "$CONFIG_DIR/config-menu" << EOF
{
    "layer": "top",
    "position": "top",
    "height": 40,
    "modules-left": ["custom/menu-toggle", "custom/app-name", "custom/app-menus"],
    "modules-center": ["clock"],
    "modules-right": ["cpu", "memory", "network", "pulseaudio", "battery", "tray"],
    
    "custom/menu-toggle": {
        "format": " Menu",
        "on-click": "$CONFIG_DIR/modules/menu-toggle.sh"
    },
    
    "custom/app-name": {
        "exec": "echo $APP",
        "format": " {}",
        "interval": 1
    },
    
    "custom/app-menus": {
        "exec": "$CONFIG_DIR/modules/get-app-menus.sh",
        "format": "{}",
        "return-type": "json",
        "interval": 1
    }
}
EOF
    
    # Reload waybar with menu config
    pkill waybar
    waybar -c "$CONFIG_DIR/config-menu" &
else
    # Switch back to workspaces view
    echo "workspaces" > "$STATE_FILE"
    
    # Reload waybar with normal config
    pkill waybar
    waybar &
fi
